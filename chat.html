# Ejemplo de chat bot con listas b√°sico
import random

# Lista de saludos y respuestas
saludos = ["hola", "que tal", "buenos dias", "buenas tardes", "buenas noches"]
respuestas = ["hola, como estas?",
            "como puedo ayudarte",
            "hola como te va",
            "que hay?"]

despedida = ["adios", "bye", "gracias", "nos vemos", "hasta luego", "salir"]
respuesta_despedida = [
    "adios, hasta pronto",
    "fue un placer ayudarte",
    "cuidate",
    "nos vemos luego",
    "bye"
]

pregunta = ["que haces"]
respuesta_pregunta = ["ayudandote a contestar tus preguntas",
                  "nada estoy triste",
                  " estoy enojado",
                  " estoy feliz"]

# Mensaje de bienvenida del chatbot
print("Hola soy Chato, tu chabot de confianza, si deseas "
      "terminar escribe salir")

while True:
    usuario = input("Tu:").lower().strip()

    if usuario in saludos:
        print("chatbot:", random.choice(respuestas))
    elif usuario in pregunta:
        print("chatbot:", random.choice(respuesta_pregunta))
    elif usuario in despedida:
        print("chatbot:", random.choice(respuesta_despedida))
        break

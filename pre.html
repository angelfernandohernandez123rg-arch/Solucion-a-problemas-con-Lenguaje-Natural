NLTK (Natural Language Toolkit) es una biblioteca poderosa para procesamiento de lenguaje natural. Permite tokenizaci√≥n avanzada por palabras y oraciones, con soporte para m√∫ltiples idiomas. 

import re
import nltk

# Descarga de modelo de tokenizaci√≥n
# nltk.download('punkt_tab')

# importamos las funciones de la librer√≠a
from nltk.tokenize import word_tokenize, sent_tokenize

# ejemplo de texto a tokenizar
texto = ("En la materia de lenguaje natural PLN y visi√≥n artificial VA, se desarrolla "
         "un portafolio de pr√°cticas como material de apoyo para los siguientes "
         "semestres de la especialidad, incluyendo pr√°cticas y glosario de funciones "
         "por cada pr√°ctica")

# Como buena pr√°ctica al tokenizar, hay que limpiar el texto
# Convertir en min√∫scula
texto_limpio = texto.lower()

# Quitar saltos de l√≠nea y espacios
texto_limpio = texto_limpio.strip()

# Elimina caracteres que no sean letras
texto_limpio = re.sub(r'[^a-z√°√©√≠√≥√∫√º√±\s]', '', texto_limpio)

# TOKENIZAR
# Token por palabras
palabras = word_tokenize(texto_limpio, language='spanish')
print("Token en palabras:", palabras)

# Token por oraciones
oraciones = sent_tokenize(texto)
print("Token en oraciones:", oraciones)

print("Texto original:", texto)
print("Texto Limpio:", texto_limpio)

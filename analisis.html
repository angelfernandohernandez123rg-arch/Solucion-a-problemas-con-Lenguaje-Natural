El an√°lisis de sentimientos permite identificar la polaridad emocional de un texto, clasific√°ndolo como positivo, negativo o neutral mediante el conteo de palabras clave. 

# Definir listas de palabras emocionales
positivas = ["feliz"]
negativas = ["perdido", "frustrado", "triste", "insoportablemente", "confundido"]

# Texto a analizar
texto = ("feliz feliz perdido")

# Tokenizaci√≥n y normalizaci√≥n

conteoposi = 0
conteonega = 0

# An√°lisis de sentimientos

    if palabra in positivas:
         conteoposi += 1
    elif palabra in negativas:
        conteonega  += 1

# Resultados
print("Palabras positivas encontradas:", conteoposi)
print("Palabras negativas encontradas:", conteonega)

# Clasificaci√≥n final
if conteonega > conteoposi:
    print("La frase es negativa.")
elif conteoposi > conteonega:
    print("La frase es positiva.")
